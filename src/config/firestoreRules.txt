service cloud.firestore {
  match /databases/{database}/documents {
    
    // Función para verificar si el usuario logueado es admin
    // Esto busca el documento del usuario en la colección 'users'
    function esAdmin() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.rol == "admin";
    }

    // Colección de productos: Público para leer, restringido para editar
    match /products/{productId} {
      allow read: if true;
      // Permitir escribir si es admin, O si solo está modificando el stock (reduciéndolo)
      allow update: if request.auth != null && (
        esAdmin() || 
        request.resource.data.stock < resource.data.stock // Solo permite si el stock nuevo es menor al viejo
      );
      allow create, delete: if request.auth != null && esAdmin();
    }

    // Colección de mensajes: Cualquiera puede enviar, pero solo el Admin puede leer o gestionar
    match /messages/{messageId} {
      allow create: if true; // 🌟 Cualquiera en la web puede usar el formulario para mandar un mensaje
      allow read, update, delete: if request.auth != null && esAdmin(); // 🔒 Solo vos desde el Dashboard podés verlos o borrarlos
    }

    // Colección de usuarios: Solo el propio usuario puede leer/editar su perfil
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Colección de carritos: Solo el dueño del carrito puede acceder
    match /carts/{cartId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    // Colección de pedidos: Solo el dueño de la orden puede leerla
    match /orders/{orderId} {
      // Permitimos lectura si el usuario está logueado Y 
      // el campo 'buyerUid' en la orden coincide con su ID
      allow read: if request.auth != null && request.auth.uid == resource.data.buyerUid;
      
      // Permitimos crear si el usuario está logueado
      allow create: if request.auth != null;
      
      // Permitimos a admins leer/escribir todo
      allow write: if request.auth != null && esAdmin();
    }
  }
}
